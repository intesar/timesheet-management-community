#drop trigger user_phone_bu;

CREATE trigger USER_PHONE_BU

  before update on USER_PHONE

  for each row

    insert into USER_PHONE_AUDIT ( USER_ID, USERNAME, PHONE_TYPE, PHONE_NUMBER, IS_PUBLIC,
      CREATION_DATE, IS_DELETED, AUDIT_USER, AUDIT_DATE )
        values ( OLD.ID, OLD.USERNAME, OLD.PHONE_TYPE, OLD.PHONE_NUMBER, OLD.IS_PUBLIC,
           OLD.CREATION_DATE, OLD.IS_DELETED, OLD.LAST_MODIFIED_BY, CURRENT_DATE );
